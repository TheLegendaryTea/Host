use Bt_Service
        CREATE TABLE Users (
            id int PRIMARY KEY,
            first_name VARCHAR(100),
          last_name VARCHAR(100),
          patronymic  VARCHAR(100),
            phone VARCHAR(20),
            login VARCHAR(50) UNIQUE,
            password VARCHAR(50),
            type VARCHAR(20) CHECK (type IN ('Менеджер', 'Мастер', 'Оператор', 'Заказчик'))
        );
        CREATE TABLE Requests (
            id int PRIMARY KEY,
            status VARCHAR(30) CHECK (status IN ('новая заявка','в процессе ремонта','готова к выдаче')),
            completion_date DATE,
            repair_parts TEXT,
            master_id INT REFERENCES Users(id),
            client_id INT REFERENCES Users(id)
        );
        CREATE TABLE Comments (
            id int PRIMARY KEY,
            message TEXT,
            master_id INT REFERENCES Users(id),
            request_id INT REFERENCES Requests(id)
        );
        CREATE TABLE Parts (
            id int PRIMARY KEY,
            name VARCHAR(100),
            stock_quantity INT,
            price NUMERIC
        );
        CREATE TABLE RequestParts (
            id int PRIMARY KEY,
            request_id INT REFERENCES Requests(id),
            part_id INT REFERENCES Parts(id),
            quantity INT
        );
